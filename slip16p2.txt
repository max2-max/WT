<html>
<head><script>
function show(str)
{
var x=new XMLHttpRequest();
x.onreadystatechange=function()
{
if(x.readyState==4 && x.status==200)
{
var xml=x.responseXML;
var b=xml.getElementsByTagName("book");
for(var i=0;i<b.length;i++)
{
var t=b[i].getElementsByTagName("title")[0].childNodes[0].nodeValue;
if(t==str)
{
document.getElementById("d").innerHTML=
"Author:"+b[i].getElementsByTagName("author")[0].childNodes[0].nodeValue+
"<br>Year:"+b[i].getElementsByTagName("year")[0].childNodes[0].nodeValue+
"<br>Price:"+b[i].getElementsByTagName("price")[0].childNodes[0].nodeValue;
}}
}
};
x.open("GET","slip16.xml",true);
x.send();
}
</script>
</head>

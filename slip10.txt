<html>
<head>
<title>jQuery before() and after()</title>
<script
src="https://ajax.googleapis.com/ajax/libs/jquery/3.6.0/jquery.min.js"></script><script>
$(document).ready(function(){
$("#btn1").click(function(){
$("#para").before("<p>This text is inserted BEFORE the paragraph.</p>");
});
$("#btn2").click(function(){
$("#para").after("<p>This text is inserted AFTER the paragraph.</p>");
});
});
</script>
</head>
<body>
<h2>jQuery before() and after() Example</h2>
<p id="para">This is the original paragraph.</p>
<button id="btn1">Insert Before</button><button id="btn2">Insert After</button>
</body>
